                  # 并查集

                  # 问题背景

                  需要维护数据之间的关系，例如A是B小弟，B是C小弟，E是D的大哥，那么这5个人有几个派系呢？ 要想维护这个关系，就需要使用并查集来操作。

                  //假设刚开始有5个人，用一个大小为5的数组保存自己老大的信息
//刚开始派系数等于总人数，count=5
//每次有个人被俘获了，派系就少1个 所以可以在找关系的时候控制新的派系
int[] boss=new int[5];
//在关系还没有发生的时候，每个人的老大都是自己
//===初始化===
boss[0]=0;
boss[1]=1;
boss[2]=2;
boss[3]=3;
boss[4]=4;
//===找关系===
//这个时候小A从了B,那么A的老大就要更新成为B
boss[0]=1;
count--;
//这个时候小B从了C，那么B的老大就要更新成C
boss[1]=2;
count--;
//同理
boss[3]=4;
count--;
//这里发现A B 都从了C，D从了E，这个时候count=2，所以就只有2个派系

                  # 基础代码描述

                  
class UnionFind{
  //保存派系数目
  int count;
  //保存每个人的老大信息
  int[] boss;
  UnionFind(int size){
    boss=new int[size];
    count=size;
    for(int i=0;i<size;i++){
        boss[i]=i;
    }
  }

  //找i的老大 
  //可以发现 真正的老大都是 boss[k]=k的，所以要循环一下
  int find(int i){
    while(i! =boss[i]){//i 自己不是老大
      i=boss[i];  //把直接老大叫过来 让他去找他的老大，直到找到为止
    }
    return boss[i];
  }

  //假设 i 从了 j,
  void union(int i,int j){
    //如果i和j已经是一伙的了，那就不合并了
    int bossi=find(i);
    int bossj=find(j);
    if(bossi == bossj){
      return;
    }
    //不是一伙的，那就让i的老大 从了j
    boss[bossi]=bossj;
    //再把派系--
    count--;
  }
}

                  # 实战

                  547. 省份数量open in new window

                  有 n 个城市，其中一些彼此相连，另一些没有相连。如果城市 a 与城市 b 直接相连，且城市 b 与城市 c 直接相连，那么城市 a 与城市 c 间接相连。

省份 是一组直接或间接相连的城市，组内不含其他没有相连的城市。

给你一个 n x n 的矩阵 isConnected ，其中 isConnected[i][j] = 1 表示第 i 个城市和第 j 个城市直接相连，而 isConnected[i][j] = 0 表示二者不直接相连。

返回矩阵中 省份 的数量。

                  思路: 这个一看就是典型的维护多个城市之间的关系，最后求一个几个省份。 可以直接套用上面的基础模版，直接返回count就可以

                  public int findCircleNum(int[][] isConnected) {
    int n=isConnected.length;
    UnionFind unionFind=new UnionFind(n);
    for(int i=0;i<n;i++){
        for(int j=i+1;j<n;j++){
            if(isConnected[i][j]==1){
                unionFind.union(i,j);
            }
        }
    }
    return unionFind.count;
}

                  # 代码优化 路径压缩

                  在关系构建完成之后，我们会发现，要去找最大的老大那里，也就是find函数耗时较高，可以优化一下

                  //比如boss[A]=B,boss[B]=C,boss[C]=D，在find A的过程中，
//可以把boss[A]的上级直接换成C，这样在下次find的时候就会少一次循环
//非递归方式
int find(int i){
    while(i! =boss[i]){
      boss[i]=boss[boss[i]];
      i=boss[i];
    }
    return boss[i];
  }

//递归方式 这样好看一点
int find(int i){
  if(i != boss[i]){
    boss[i]=find(boss[i]);
  }
  return boss[i];
}

                  # 代码优化 按照秩压缩

                  每个人都配一个秘书，随时统计下面有多少层的小弟，每次合并的时候，总是把下面小弟层数少的合并到小弟层数多的上面去

                  
class UnionFind{
  //保存派系数目
  int count;
  //保存每个人的老大信息
  int[] boss;
  //保存下面小弟层数的信息
  int[] level;

  UnionFind(int size){
    boss=new int[size];
    level=new int[size];
    count=size;
    for(int i=0;i<size;i++){
        boss[i]=i;
        level[i]=1;
    }
  }

  //找i的老大 
  //可以发现 真正的老大都是 boss[k]=k的，所以要循环一下
  int find(int i){
    while(i! =boss[i]){//i 自己不是老大
      i=boss[i];  //把直接老大叫过来 让他去找他的老大，直到找到为止
    }
    return boss[i];
  }

  //比较i和j的小弟层数，少的加到多的上面去
  void union(int i,int j){
    //如果i和j已经是一伙的了，那就不合并了
    int bossi=find(i);
    int bossj=find(j);
    if(bossi == bossj){
      return;
    }
    if(level[i]>level[j]){
      //j 加到 i 上去
      boss[bossj]=bossi;
      level[i]+=level[j];//j 的层数都给i
    }else{
      boss[bossi]=bossj;
      level[j]+=level[i];
    }
    count--;
  }
}
